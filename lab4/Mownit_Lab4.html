# mnoenie macierzy - wersja naiwna
function naive_multiplication(A,B)
    C = zeros(Float64, size(A,1), size(B,2))
    for i=1:size(A,1)
        for j=1:size(B,2)
            for k=1:size(A,2)
                C[i,j] = C[i,j] + A[i,k]*B[k,j]
            end
        end
    end
    C
end

# poprawiona funkcja korzytajca z powyszego oraz z faktu, e
# mona zmienia kolejno operacji dodawania (a co za tym idzie kolejnosc petli).
function better_multiplication(A, B)
    C = zeros(Float64, size(A,1), size(B,2))
    for j=1:size(B,2)
        for k=1:size(A,2)
            for i=1:size(A,1)
                C[i,j] = C[i,j] + A[i,k]*B[k,j]
            end
        end
    end
    C
end

Zadania¶

  1. Uruchomi

    • naive_multiplication(A,B),
    • better_multiplication(A,B)

    • mnoenie BLAS w Julii (A*B)

      dla coraz wikszych macierzy kwadratowych i zmierzy czasy. Narysowa wykres zaleynoci czasu od rozmiaru macierzy wraz z supkami bdw, tak jak na poprzednim laboratorium. Wszystkie trzy metody powinny by na jednym wykresie. (1pkt)

  1. Napisa w jzyku C:

    • naiwn metod mnoenia macierzy kwadratowych (wersja 1)
    • ulepszon wersj za pomoc zamiany ptli metod mnoenia macierzy(wersja 2), pamitajc, e w C macierz przechowywana jest wierszami (row major order tzn A11,A12, ..., A1m, A21, A22,...,A2m, ..Anm), inaczej ni w Julii !

    • skorzysta z moliwoci BLAS poziom 3 dostpnego w GSL (przykad uzycia https://www.gnu.org/software/gsl/doc/html/blas.html#examples )

      Naley porwnywa dziaanie tych trzech algorytmow bez wczonej opcji optymalizacji kompilatora. Przedstawi wyniki na jednym wykresie tak jak w p.1. (osobno ni p.1). (1 pkt)

      (Dla chtnych) sprawdzi, co si dzieje, jak wczymy optymalizacj kompilatora i doda do wykresu.

  1. Uy funkcji polyfit z pakietu Polynomials do znalezienia odpowiednich wielomianw, ktre najlepiej pasuj do zalenoci czasowych kadego z algorytmw. Stopie wielomianu powinien zgadza si z teoretyczn zoonosci. Doda wykresy uzyskanych wielomianw do wczeniejszych wykresw. (1 pkt)

  2. Pokaza zniwelowanie efektu Rungego (przy interpolacji) poprzez uycie wsparcia dla wielomianw Czebyszewa w pakiecie Polynomials. Narysowa wybran funkcj z zaznaczonymi wzami i wielomianem interpolacyjnym dla wzw rwnoodlegych oraz Czebyszewa (2 wykresy).(1 pkt)

  3. Przyblienie Pade jest czsto lepsze ni rozwinicie w szereg Taylora przy aproksymowaniu funkcji, ktre posiadaj osobliwoci. Korzystajc ze wsparcia dla aproksymacji Pade w pakiecie Polynomials pokaza dowolny przykad (wraz z wykresem), gdzie takie przyblienie faktycznie jest lepsze. Mona odtworzy wykres z wykadu albo zainspirowa si przykadowym artykuem https://www.hindawi.com/journals/ijcm/2014/587430/ (1 pkt)
